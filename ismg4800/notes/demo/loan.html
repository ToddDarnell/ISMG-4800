<script language="JavaScript">
 function Minterest(interest)
 {
    return (interest/100)/12;
 }

 function Payment(yri, amt, dur)
 {
   var r = Minterest(yri);
   return (r*amt*Math.pow((1+r),dur)/(Math.pow((1+r),dur)-1));
 }

 function TotalInterest (yri, amt, dur)
 {
   var payment = Payment(yri, amt, dur);
   return dur*payment-amt ;
 }
 
 function F2(num)
 {
    return Math.round(num*100)/100;
 }

 function Loan()
 {
   window.status = "Calculating Loan Payment";
   var yri = parseFloat(document.loanForm.interest.value);
   var dur = parseInt(document.loanForm.duration.value);
   var amt = parseInt(document.loanForm.amount.value);
 
   document.loanForm.payment.value = F2(Payment(yri, amt, dur));
   document.loanForm.total.value = F2(TotalInterest(yri, amt, dur));
 
   window.status = "Loan Calculation Complete";
 }
