    function checkSelection(sel)
    {
        var message = "The " + sel.name + " list has " + sel.length + " options\n";
        message += "The list has option " + sel.selectedIndex + " selected first\n";
        message += "The first selected option is " + sel.options[sel.selectedIndex].text +"\n";
        message += "The first selected option's value is " + sel.options[sel.selectedIndex].value  +"\n\n";
        
        if(sel.multiple)
        {
           for(var i=0; i < sel.length; i++)
           {
             if(sel.options[i].selected)
             {
               message += "List options[" +i+ "].text is " + sel.options[i].text +"\n";
               message += "List options[" +i+ "].value is " + sel.options[i].value +"\n";
             }
           }
        }
        alert(message);
    }
