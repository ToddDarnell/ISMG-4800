<script language="JavaScript">
  var marcode=prompt("Enter Marital Status (M, S, D, W)");
  switch(marcode)
  {
    case 'M':
      document.write("<p>Individual is married.</p>");
      break;
    case 'S':
      document.write("<p>Individual is single.</p>");
      break;
    case 'D':
      document.write("<p>Individual is divorced.</p>");
      break;
    case 'W':
      document.write("<p>Individual is widowed.</p>");
      break;
    default:
      document.write("<p>Invalid marriage code.</p>");
  }
</script>
