<script language="JavaScript"> 
   var temp=prompt('Enter a temperature (F)');
   if (temp < 35) {
       document.write('<h1>' + temp + ' is cold!</h1>'); 
  }
  else if (temp < 60) {
        document.write('<h1>'+ temp +' is chilly!</h1>'); 
  }
  else if (temp < 90) {
        document.write('<h1>'+ temp +' is warm!</h1>'); 
  }
  else {
        document.write('<h1>'+ temp +' is hot!</h1>'); 
  }
</script>
